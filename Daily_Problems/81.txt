class Solution {

    /*
     * Problem Summary:
     * We are given a rotated sorted array that may contain duplicates.
     * We must search for a target value and return true if it exists, otherwise false.
     * 
     * Key Idea:
     * Use a modified Binary Search:
     * - Even if the array is rotated, at least one side (left half or right half) will be sorted.
     * - Compare target with the sorted side to decide which direction to go.
     * - If duplicates make it unclear which side is sorted, shrink the search space by moving pointers.
     * 
     * Worst Case Time Complexity:
     * - O(n), because duplicates can force us to reduce search space one-by-one.
     * - Best/Average remains close to O(log n).
     */

    public boolean search(int[] nums, int target) {
        int left = 0;
        int right = nums.length - 1;

        while (left <= right) {

            int mid = left + (right - left) / 2;

            // Case 1: Found the target directly
            if (nums[mid] == target) {
                return true;
            }

            // Case 2: When duplicates prevent us from knowing the sorted half
            // Example: [3, 3, 3, 3, 3] or [2, 5, 6, 0, 0, 1, 2]
            // If left, mid, and right are equal, we cannot decide which half is sorted.
            // So we simply shrink the search space.
            if (nums[left] == nums[mid] && nums[mid] == nums[right]) {
                left++;
                right--;
            }

            // Case 3: Left half is sorted
            else if (nums[left] <= nums[mid]) {
                // Check if target lies inside the sorted left half
                if (nums[left] <= target && target < nums[mid]) {
                    // If yes, move search to left half
                    right = mid - 1;
                } else {
                    // Otherwise, move to right half
                    left = mid + 1;
                }
            }

            // Case 4: Right half is sorted
            else {
                // Check if target lies inside the sorted right half
                if (nums[mid] < target && target <= nums[right]) {
                    // Move search to right half
                    left = mid + 1;
                } else {
                    // Otherwise move to left half
                    right = mid - 1;
                }
            }
        }

        // If loop ends without finding target -> not present
        return false;
    }
}
