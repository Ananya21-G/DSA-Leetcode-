/*
üéØ Problem: Sorting the Sentence
--------------------------------
Given a shuffled sentence where each word ends with a digit (its original position),
reconstruct and return the original sentence.

Example:
Input:  "is2 sentence4 This1 a3"
Output: "This is a sentence"
*/

class Solution {
    public String sortSentence(String s) {
        // Step 1Ô∏è‚É£: Split the sentence into words
        String[] words = s.split(" ");

        // Step 2Ô∏è‚É£: Create an array to hold words in their correct order
        String[] arranged = new String[words.length];

        // Step 3Ô∏è‚É£: Process each word to find its correct position
        for (String word : words) {
            // Extract the last character (the position number)
            int index = word.charAt(word.length() - 1) - '0'; // Convert '1' ‚Üí 1, '2' ‚Üí 2, etc.

            // Extract the actual word (remove the last character)
            String actualWord = word.substring(0, word.length() - 1);

            // Store the word at correct position (subtract 1 for 0-based indexing)
            arranged[index - 1] = actualWord;
        }

        // Step 4Ô∏è‚É£: Join all words back into a sentence
        return String.join(" ", arranged);
    }
}
