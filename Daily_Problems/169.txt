class Solution {
    public int majorityElement(int[] nums) {

        // candidate will store the possible majority element
        int candidate = 0;

        // count tracks the "vote balance" for the current candidate
        int count = 0;

        // Traverse each number in the array
        for (int num : nums) {

            // If count is 0, we pick the current number as the new candidate
            // (because the previous candidate got fully cancelled out)
            if (count == 0) {
                candidate = num;
            }

            // If current number matches candidate, give it a vote (+1)
            if (candidate == num) {
                count++;
            }
            // Otherwise, cancel one vote (-1)
            else {
                count--;
            }
        }

        // Since majority element is guaranteed to exist,
        // the final candidate is the answer
        return candidate;
    }
}
